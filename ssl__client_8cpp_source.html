404  void SendError(std::error_code ec)
405  {
406  // Skip Asio disconnect errors
407  if ((ec == asio::error::connection_aborted) ||
408  (ec == asio::error::connection_refused) ||
409  (ec == asio::error::connection_reset) ||
410  (ec == asio::error::eof) ||
411  (ec == asio::error::operation_aborted))
412  return;
413 
414  // Skip OpenSSL annoying errors
415  if (ec == asio::ssl::error::stream_truncated)
416  return;
417  if (ec.category() == asio::error::get_ssl_category())
418  {
419  if ((ERR_GET_REASON(ec.value()) == SSL_R_DECRYPTION_FAILED_OR_BAD_RECORD_MAC) ||
420  (ERR_GET_REASON(ec.value()) == SSL_R_PROTOCOL_IS_SHUTDOWN) ||
